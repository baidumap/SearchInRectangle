 24     /**

 26      */
 27     var SearchInRectangle =
 28         /**
 29          * SearchInRectangle类的构造函数
 30          * @class 拉框搜索类，实现拉框搜索效果的<b>入口</b>。
 31          * 实例化该类后，即可调用该类提供的open
 32          * 方法开启拉框搜索状态。
 33          * 
 34          * @constructor
 35          * @param {Map} map Baidu map的实例对象
 36          * @param {String} keyword 搜索关键字 
 37          * @param {Json Object} opts 可选的输入参数，非必填项。可输入选项包括：<br />         
 38          * {<dl><dt>"<b>renderOptions</b>" : {Json Object} 显示设置选项，可选</dt>
 39          * <dd>{</dd>
 40          * <dd>"<b>map</b>" : {Map} 搜索结果显示设置</dd>
 41          * <dd>"<b>followText</b>" : {String} 开启拉框搜索状态后，鼠标跟随的文字</dd>
 42          * <dd>"<b>strokeWeight</b>" : {Number} 遮盖层外框的线宽</dd>
 43          * <dd>"<b>strokeColor</b>" : {String} 遮盖层外框的颜色</dd>
 44          * <dd>"<b>style</b>" : {String} 遮盖层外框的样式</dd>
 45          * <dd>"<b>opacity</b>" : {Number} 遮盖层的透明度</dd>
 46          * <dd>"<b>cursor</b>" : {String} 鼠标样式<br/> </dd>
 47          * <dd>"<b>autoClose</b>" : {Boolean} 拉框完成后是否自动关闭 <br/> </dd> 
 48          * <dd>"<b>alwaysShowOverlay</b>" : {Boolean} 是否一直显示拉框的覆盖物 <br/> </dd> 
 49          * <dd>"<b>panel</b>" : {String|HTMLElement} 结果列表的HTML容器id或容器元素，提供此参数后，结果列表将在此容器中进行展示。 <br/> </dd>
 50          * <dd>"<b>selectFirstResult</b>" : {Boolean} 是否选择第一个检索结果。 <br/> </dd>
 51          * <dd>},</dd>
 52          * <dt>"<b>onSearchComplete</b>" : {Function} 可选，搜索完成后调用此方法 </dt></dl>
 53          * }
 54          * @example <b>参考示例：</b><br />
 55          * var map = new BMap.Map("container");<br />map.centerAndZoom(new BMap.Point(116.404, 39.915), 15);<br />var searchInRect = new BMapLib.SearchInRectangle(map,"餐馆",{strokeWeight : 2});
 56          */
 57         BMapLib.SearchInRectangle = function(map, searchWords ,opts){
 58             if (!map) {
 59                 return;
 60             }
 61 
 62             /**
 63              * map对象
 64              * @private
 65              * @type {Map}
 66              */
 67             this._map = map;
 68             
 69             /**
 70              * 要搜索的东西，例如“餐馆”
 71              * @private
 72              * @type {String}
 73              */
 74             this._searchWords = searchWords ;
 75             /**
 76              * 各种状态的默认参数
 77              * @private
 78              * @param {Json Object} 
 79              */
 80             this._opts = {
 81                 //搜索结果显示设置
 82                 map: map,                
 83                 // 开启拉框搜索状态后，鼠标跟随的文字
 84                 followText : "",
 85                 // 遮盖层外框的线宽
 86                 strokeWeight : 2,
 87                 // 遮盖层外框的颜色
 88                 strokeColor : "#111",
 89                 // 遮盖层外框的样式
 90                 style : "solid",
 91                 // 遮盖层的填充色
 92                 fillColor : "#ccc",
 93                 // 遮盖层的透明度
 94                 opacity : 0.4,
 95                 // 鼠标样式
 96                 cursor : "crosshair",
 97                 // 是否在每次操作后，自动关闭拉框搜索状态, 私有属性
 98                 autoClose : true,                
 99                 //是否自动调整视野
100                 autoViewport : false,
101                 //是否一直显示拉框后的覆盖物
102                 alwaysShowOverlay: true,
103                 //显示面板
104                 panel:"",
105                 //是否显示第一个搜索结果
106                 selectFirstResult: "false",
107                 // 拉框后放大
108                 _zoomType : BMAP_ZOOM_IN
109             };
110             
111             // 通过使用者输入的opts，修改这些默认参数
112             this._setOptions(opts);
113 
114             // 验证参数正确性
115             this._opts.strokeWeight = 
116                 this._opts.strokeWeight <= 0 ? 
117                     1 : 
118                     this._opts.strokeWeight;
119 
120             this._opts.opacity = 
121                 this._opts.opacity < 0 ? 
122                     0 : 
123                     this._opts.opacity > 1 ? 
124                         1 : 
125                         this._opts.opacity;
126 
127             /**
128              * 当前是否开启拉框搜索状态；默认为false，表示没有开启
129              * @private
130              * @type {Boolean} 
131              */
132             this._isOpen = false;
133 
134             /**
135              * 拉框时显示的矩形遮盖层
136              * @private
137              * @type {HTMLElement}
138              */
139             this._fDiv = null;
140             
141             /**
142              * 鼠标跟随的文字提示框
143              * @private
144              * @type {BMap.Label}
145              */
146             this._followTitle = null;
147             //当前overlay对象
148             this._overlay = null;
149             //创建一个BMap.LocalSearch()实例,并赋值给local
150             this.local = this.localSearch(this);            
151         }

153     /**
154      * 根据用户输入的opts，修改默认参数_opts
155      * @param {Json Object} opts 用户输入的修改参数
156      * @return 无返回值
157      */
158     SearchInRectangle.prototype._setOptions = function(opts) {
159         if (!opts) {
160             return;
161         }
162         if(opts.renderOptions){          
163             for (var p in opts.renderOptions) {                  
164                 if (typeof(opts.renderOptions[p]) != "undefined") {
165                     this._opts[p] = opts.renderOptions[p];                   
166                 }
167             }
168         }
169         if(opts.onSearchComplete){          
170             this._opts["onSearchComplete"] = opts.onSearchComplete ;
171         }
172     };

277             var st = this._fDiv.style,
278                   op = this._opts.opacity;
279             st.opacity = op; 
280             st.MozOpacity = op;
281             st.KhtmlOpacity = op;
282             st.filter = "alpha(opacity=" + (op * 100) + ")";
283         }

297     /**
298      * 控件项的事件绑定
299      * @return 无返回值
300      */
301     SearchInRectangle.prototype._bind = function(){
302         // 设置鼠标样式
303         this.setCursor(this._opts.cursor);
304         var me = this;
305         // 在装载地图的页面元素上，绑定鼠标移动事件
306         addEvent(this._map.getContainer(), "mousemove", function(e){
307             if (!me._isOpen) {
308                 return;
309             }
310             if (!me._followTitle) {
311                 return;
312             }
313             e = window.event || e;
314             var t = e.target || e.srcElement;
315             // 如果触发该事件的页面元素不是遮盖效果层，则返回，无操作
316             if (t != OperationMask.getDom(me._map)) {
317                 me._followTitle.hide();
318                 return;
319             }
320             if (!me._mapMoving) {
321                 me._followTitle.show();
322             }
323             // 设置鼠标移动过程中，跟随的文字提示框的位置
324             var pt = OperationMask.getDrawPoint(e, true);
325             me._followTitle.setPosition(pt);
326         });
327         // 创建鼠标跟随的文字提示框
328         if (this._opts.followText) {
329             var t = this._followTitle = new BMap.Label(this._opts.followText, {offset : new BMap.Size(14, 16),enableMassClear: false});
330             this._followTitle.setStyles({color : "#333", borderColor : "#ff0103"});
331         }
332     };

334     /**
335      * 开启拉框搜索状态。
336      * 在拉框效果结束的时候，会调用Animation库(见源文件，闭包，不对外开放)
337      * 来实现一些小动画
338      * @return 成功开启拉框搜索状态时，返回true；否则无返回值
339      *
340      * @example <b>参考示例：</b><br />
341      * searchInRect.open();
342      */
343     SearchInRectangle.prototype.open = function() {
344         // 判断拉框搜索状态是否已经开启
345         if (this._isOpen == true) {
346             return true;
347         }
348         // 已有其他地图上的鼠标操作工具开启
349         if (!!BMapLib._toolInUse) {
350             return;
351         }     
352         BMapLib._toolInUse = true;
353 
354         this._isOpen = true;
355 
356         // 增加鼠标在地图区域移动的事件
357         // 通过binded参数，避免多次绑定
358         if (!this.binded) {
359             this._bind();
360             this.binded = true;
361         }
362 
363         // 将文字提示框作为BMap.Label元素，提交给Map Api进行管理
364         if (this._followTitle) {
365             this._map.addOverlay(this._followTitle);
366             this._followTitle.hide();
367         }
368 
369         var me = this;
370         var map = this._map;
371         // 返回IE版本号
372         var ieVersion = 0;
373         if (/msie (\d+\.\d)/i.test(navigator.userAgent)) {
374             ieVersion = document.documentMode || + RegExp['\x241']
375         }
376         
377         var moveDistance = 0 ;//移动距离，如果是0，说明进行的是点击操作
378         // 开启拉框搜索状态后，鼠标在地图上按下时的操作
379         var beginDrawRect = function(e) {        
380             // 由于在IE和非IE浏览器下，e对象对鼠标按下键的返回值不一样
381             // 所以需要分两种情况判断，当不是鼠标左键时，返回无操作
382             e = window.event || e;
383             if (e.button != 0 && 
384                 !ieVersion ||
385                 ieVersion && 
386                 e.button != 1) {
387                     return;
388             }
389 
390             // 增加IE浏览器下对事件的捕获
391             if (!!ieVersion && OperationMask.getDom(map).setCapture) {
392                 OperationMask.getDom(map).setCapture();
393             }
394                 
395             if (!me._isOpen) {
396                 return;
397             }
398             moveDistance = 0 ; 
399             me._bind.isZooming = true;
400             
401             // 添加拖拽鼠标画框时，鼠标移动事件，和鼠标弹起事件
402             addEvent(document, "mousemove", drawingRect);
403             addEvent(document, "mouseup", endDrawRect);
404 
405             // 记录此时鼠标相关位置
406             me._bind.mx = e.layerX || e.offsetX || 0;
407             me._bind.my = e.layerY || e.offsetY || 0;
408             me._bind.ix = e.pageX || e.clientX || 0;
409             me._bind.iy = e.pageY || e.clientY || 0;
410 
411             // 创建矩形半透明效果框
412             insertHTML(OperationMask.getDom(map), "beforeBegin", me._generateHTML());
413             me._fDiv = OperationMask.getDom(map).previousSibling;
414 
415             me._fDiv.style.width = "0";
416             me._fDiv.style.height = "0";
417             me._fDiv.style.left = me._bind.mx + "px";
418             me._fDiv.style.top = me._bind.my + "px";
419 
420             // 停止事件冒泡传播和默认事件
421             stopBubble(e);
422             return preventDefault(e);
423         };
424 
425         // 开启拉框搜索状态后，鼠标在地图上按下、并拖拽时的操作        
426         var drawingRect = function(e) {  
427             if (me._isOpen == true && me._bind.isZooming == true) {
428 
429                 //再次拉框的时候，如果地图上还有其他框存在，则先清除掉
430                 if(me._overlay){ me._map.removeOverlay(me._overlay); }
431                 // 通过鼠标当前所在位置，计算矩形半透明效果框的高宽
432                 var e = window.event || e;
433                 var curX = e.pageX || e.clientX || 0;
434                 var curY = e.pageY || e.clientY || 0;
435 
436                 var dx = me._bind.dx = curX - me._bind.ix;
437                 var dy = me._bind.dy = curY - me._bind.iy;
438                 var tw = Math.abs(dx) - me._opts.strokeWeight;
439                 var th = Math.abs(dy) - me._opts.strokeWeight;
440                 me._fDiv.style.width = (tw < 0 ? 0 : tw) + "px";
441                 me._fDiv.style.height = (th < 0 ? 0 : th) + "px";
442                 moveDistance = tw;
443                 // 计算矩形半透明效果框所在位置
444                 var mapSize = [map.getSize().width, map.getSize().height];
445                 // 当dx小于0的时候，也就是绘制中的点位置，在水平方向上，比起始点更靠左
446                 // 说明，此时用户在从右往左绘制矩形框
447                 // 需要对矩形的右边距进行计算，与普通状况下的计算左边距，有所不同
448                 if (dx >= 0) {
449                     me._fDiv.style.right = "auto";
450                     me._fDiv.style.left = me._bind.mx + "px";
451                     if (me._bind.mx + dx >= mapSize[0] - 2 * me._opts.strokeWeight) {
452                         me._fDiv.style.width = mapSize[0] - me._bind.mx - 2 * me._opts.strokeWeight + "px";
453                         me._followTitle && me._followTitle.hide();
454                     }
455                 } else {
456                     me._fDiv.style.left = "auto";
457                     me._fDiv.style.right = mapSize[0] - me._bind.mx + "px";
458                     if (me._bind.mx + dx <= 2 * me._opts.strokeWeight) {
459                         me._fDiv.style.width = me._bind.mx - 2 * me._opts.strokeWeight + "px";
460                         me._followTitle && me._followTitle.hide();
461                     }
462                 }
463                 // 当dy小于0的时候，也就是绘制中的点位置，在垂直方向上，比起始点更靠上
464                 // 说明，此时用户在从下往上绘制矩形框
465                 // 需要对矩形的下边距进行计算，与普通状况下的计算上边距，有所不同
466                 if (dy >= 0) {
467                     me._fDiv.style.bottom = "auto";
468                     me._fDiv.style.top = me._bind.my + "px";
469                     if (me._bind.my + dy >= mapSize[1] - 2 * me._opts.strokeWeight) {
470                         me._fDiv.style.height = mapSize[1] - me._bind.my - 2 * me._opts.strokeWeight + "px";
471                         me._followTitle && me._followTitle.hide();
472                     }
473                 } else {
474                     me._fDiv.style.top = "auto";
475                     me._fDiv.style.bottom = mapSize[1] - me._bind.my + "px";
476                     if (me._bind.my + dy <= 2 * me._opts.strokeWeight) {
477                         me._fDiv.style.height = me._bind.my - 2 * me._opts.strokeWeight + "px";
478                         me._followTitle && me._followTitle.hide();
479                     }
480                 }
481 
482                 // 停止事件冒泡传播和默认事件
483                 stopBubble(e);
484                 return preventDefault(e);
485             }
486         };
487 
488         // 开启拉框搜索状态后，鼠标在地图上拖拽时、弹起的操作
489         var endDrawRect = function(e) { 
490             if(!moveDistance) me._fDiv.parentNode.removeChild(me._fDiv);
491             if (me._isOpen == true) { 
492                 // 删除拖拽鼠标画框时，鼠标移动事件，和鼠标弹起事件
493                 removeEvent(document, "mousemove", drawingRect);
494                 removeEvent(document, "mouseup", endDrawRect);
495                 if(!moveDistance){ return;}
496                 
497                 // 释放IE浏览器下对事件的捕获
498                 if (!!ieVersion && OperationMask.getDom(map).releaseCapture){
499                     OperationMask.getDom(map).releaseCapture();
500                 }
501 
502                 // 计算当前矩形半透明效果框的中心点
503                 var centerX = parseInt(me._fDiv.style.left) + parseInt(me._fDiv.style.width) / 2;
504                 var centerY = parseInt(me._fDiv.style.top) + parseInt(me._fDiv.style.height) / 2;
505                 var mapSize = [map.getSize().width, map.getSize().height];
506                 if (isNaN(centerX)) {
507                     centerX = mapSize[0] - parseInt(me._fDiv.style.right) - parseInt(me._fDiv.style.width) / 2;
508                 }
509                 if (isNaN(centerY)) {
510                     centerY = mapSize[1] - parseInt(me._fDiv.style.bottom) - parseInt(me._fDiv.style.height) / 2;
511                 }
512 
513                 // 通过对比矩形和地图区域的高宽，计算需要的缩放比例
514                 var ratio = Math.min(mapSize[0] / Math.abs(me._bind.dx), mapSize[1] / Math.abs(me._bind.dy));
515                 ratio = Math.floor(ratio);
516 
517                 // 通过屏幕上的像素坐标的转化，计算矩形半透明效果框的Bound区域
518                 var px1 = new BMap.Pixel(centerX - parseInt(me._fDiv.style.width) / 2, centerY + parseInt(me._fDiv.style.height) / 2);
519                 var px2 = new BMap.Pixel(centerX + parseInt(me._fDiv.style.width) / 2, centerY - parseInt(me._fDiv.style.height) / 2);
520                
521                 var pt1 = map.pixelToPoint(px1);
522                 var pt2 = map.pixelToPoint(px2);
523                 var bds = new BMap.Bounds(pt1, pt2);
524 
525                 delete me._bind.dx;
526                 delete me._bind.dy;
527                 delete me._bind.ix;
528                 delete me._bind.iy;
529 
530                 // 计算缩放后应该所在的地图层级
531                 // 当矩形效果框和地图区域的高宽比有清晰结果时(即ratio有计算结果)，通过高宽比来计算
532                 // 由于每层级地图间的缩放比1:2，所以使用下面的计算公式
533                 // 当ratio无计算结果时，只进行普通的1个级别的缩放改变
534                 if (!isNaN(ratio)) {
535                     //当用户设置启用自动视野调整
536                     if (me._opts._zoomType == BMAP_ZOOM_IN){                  
537                         // 拉框放大的情况
538                         targetZoomLv = Math.round(map.getZoom() + (Math.log(ratio) / Math.log(2)));
539                         if (targetZoomLv < map.getZoom()){
540                             targetZoomLv = map.getZoom();
541                         }
542                     }
543                 } else{
544                     targetZoomLv = map.getZoom() + (me._opts._zoomType == BMAP_ZOOM_IN ? 1 : -1);
545                 }
546                 
547                 // 进行层级缩放，并定位新中心点
548                 var targetCenterPt = map.pixelToPoint({x : centerX, y : centerY}, map.getZoom());
549                 //如果启用自动调整视野
550                 if(me._opts.autoViewport){
551                     map.centerAndZoom(targetCenterPt, targetZoomLv);
552                 }
553 
554                 // 设置鼠标移动过程中，跟随的文字提示框的位置
555                 var pt = OperationMask.getDrawPoint(e);
556                 if (me._followTitle) {
557                     me._followTitle.setPosition(pt);
558                     me._followTitle.show();
559                 }
560                 me._bind.isZooming = false;
561                 
562                 // 缩放操作结束，删除矩形半透明框

565             }
566

568             // 如果不需要动画效果，从此处到new Animation()的过程、以及Animation的声明，均可删除
569             var southWestPoint = bds.getSouthWest(),
570                   northEastPoint = bds.getNorthEast(),
571                   southEastPoint = new BMap.Point(northEastPoint.lng, southWestPoint.lat),
572                   northWestPoint = new BMap.Point(southWestPoint.lng, northEastPoint.lat),
573                   rect = new BMap.Polygon([
574                         southWestPoint,
575                         northWestPoint,
576                         northEastPoint,
577                         southEastPoint
578                   ]); 
579             rect.setStrokeWeight(me._opts.strokeWeight);
580             rect.setStrokeOpacity(me._opts.opacity);
581             rect.setFillOpacity(me._opts.opacity);
582             rect.setStrokeColor(me._opts.strokeColor);
583             rect.setStrokeStyle(me._opts.style);
584             rect.setFillColor(me._opts.fillColor);
585             
586             me._overlay = rect;
587             map.addOverlay(rect);
588                        
589             //拉框选取完成，然后调用搜索 
590             me.local.searchInBounds(me._searchWords,bds);
591 
592             // 渐隐藏动画效果
593             if(!me._opts.alwaysShowOverlay){              
594                 new Animation({
595                         duration : 240, 
596                         fps : 20,
597                         delay : 500, 
598                         render : function(t) {
599                             var opacity = 0.3 * (1 - t);
600                             rect.setStrokeOpacity(opacity);
601                         },
602                         finish : function() {
603                             map.removeOverlay(me._overlay);
604                             rect.dispose();
605                             me._overlay = null;
606                         }
607                 });
608             }
609             // 设置为自动关闭缩放状态时，修改相关状态值
610             if (me._opts.autoClose) {
611                 setTimeout(function() {
612                     if (me._isOpen == true) {
613                         me.close();
614                     }
615                 }, 70);
616             }      
617 
618             // 停止事件冒泡传播和默认事件
619             stopBubble(e);
620             return preventDefault(e);
621         };
622                
623         OperationMask.show(this._map);
624         this.setCursor(this._opts.cursor);
625 
626         // 增加鼠标按下时，开始绘制矩形框的事件
627         // 通过判断只绑定一次，并不再删除
628         if (!this._isBeginDrawBinded) {
629             addEvent(OperationMask.getDom(this._map), "mousedown", beginDrawRect);
630             this._isBeginDrawBinded = true;
631         }
632 
633         return true;
634     };

705     /**
706      * 在目标元素的指定位置插入HTML代码，
707      * 闭包，对外不暴露
708      *
709      * @param {HTMLElement|string} element 目标元素或目标元素的id
710      * @param {String} position 插入html的位置信息
711      *     取值为beforeBegin、afterBegin、beforeEnd或afterEnd，大小写不敏感
712      * @param {String} html 要插入的html    
713      * @return {HTMLElement} 目标元素
714      */
715     function insertHTML(element, position, html) {
716         var range,begin;
717         if (element.insertAdjacentHTML) {
718             element.insertAdjacentHTML(position, html);
719         } else {
720             range = element.ownerDocument.createRange();
721             // FF下range的位置设置错误可能导致创建出来的fragment在插入dom树之后html结构乱掉
722             // 改用range.insertNode来插入html
723             position = position.toUpperCase();
724             if (position == 'AFTERBEGIN' || position == 'BEFOREEND') {
725                 range.selectNodeContents(element);
726                 range.collapse(position == 'AFTERBEGIN');
727             } else {
728                 begin = position == 'BEFOREBEGIN';
729                 range[begin ? 'setStartBefore' : 'setEndAfter'](element);
730                 range.collapse(begin);
731             }
732             range.insertNode(range.createContextualFragment(html));
733         }
734         return element;
735     }

774      * 给某页面元素添加事件，

778      * @type {String} type 需要添加的事件名
779      * @type {Function} listener 需要触发的操作

781     function addEvent(element, type, listener) {
782         if (!element) {

784         }
785         type = type.replace(/^on/i, '').toLowerCase();
786         if (element.addEventListener) {

788         } else if (element.attachEvent) {

898          */
899         getDrawPoint : function(e, n) {
900             e = window.event || e;
901             var x = e.layerX || e.offsetX || 0;
902             var y = e.layerY || e.offsetY || 0;
903             var t = e.target || e.srcElement;
904             if (t != OperationMask.getDom(this._map) && n == true) {
905                 while (t && t != this._map.getContainer()) {
906                     if (!(t.clientWidth == 0 && 
907                          t.clientHeight == 0 && 
908                          t.offsetParent && 
909                          t.offsetParent.nodeName.toLowerCase() == 'td')) {
910                             x += t.offsetLeft;
911                             y += t.offsetTop;
912                     }
913                     t = t.offsetParent;
914                 }
915             }
916 
917             if (t != OperationMask.getDom(this._map) && 
918                 t != this._map.getContainer()) {
919                     return;
920             }
921             if (typeof x === 'undefined' || 
922                 typeof y === 'undefined') {
923                     return;
924             }
925             if (isNaN(x) || isNaN(y)) {
926                 return;
927             }
928             return this._map.pixelToPoint(new BMap.Pixel(x, y));
929         },

974     /**
975      * Animation，动画效果类，
976      * 通过该类，可以实现一些延时、规律的动画效果，
977      * 闭包在文件内，对外不暴露
978      * 
979      * @constructor
980      * @param {Json Object} opts 可选的输入参数，非必填项。可输入选项包括：
981      * {"duration" : {Number} 动画时长, 单位毫秒,
982      *  "fps" : {Number} 每秒帧数,
983      *  "delay" : {Number} 延迟执行时间，单位毫秒,
984      *  "transition" : {Object} 变换效果的曲线,
985      *  "finish" : {Function} 动画结束的回调函数,
986      *  "render" : {Function} 每一帧执行的回调函数}
987      */
988     function Animation(opts) {
989         var defaultOptions = {
990             // 动画时长, 单位毫秒
991             duration: 1000,
992             // 每秒帧数
993             fps: 30,
994             // 延迟执行时间，单位毫秒
995             delay: 0,
996             // 变换效果的曲线
997             transition: Transitions.linear,
998             // 调用stop停止时的回调函数
999             onStop: function(){} 
1000         };
1001 
1002        // 修改默认参数
1003         if (opts) {
1004             for (var i in opts) {
1005                 defaultOptions[i] = opts[i];
1006             }
1007         }
1008         this._opts = defaultOptions;
1009 
1010         if (defaultOptions.delay) {
1011             var me = this;
1012             setTimeout(function() {
1013                 me._beginTime = new Date().getTime();
1014                 me._endTime = me._beginTime + me._opts.duration;
1015                 me._launch();
1016             }, defaultOptions.delay);
1017         } else {
1018             this._beginTime = new Date().getTime();
1019             this._endTime = this._beginTime + this._opts.duration;
1020             this._launch();
1021         }
1022     }
1023 
1024     /**
1025      * 动画执行过程中的操作
1026      * @return 无返回值
1027      */
1028     Animation.prototype._launch = function() {
1029         var me = this;
1030         var now = new Date().getTime();
1031 
1032         if (now >= me._endTime) {
1033             if (typeof me._opts.render == 'function') {
1034                 me._opts.render(me._opts.transition(1));
1035             }
1036             // finish()接口，时间线结束时对应的操作
1037             if (typeof me._opts.finish == 'function') {
1038                 me._opts.finish();
1039             }
1040             return;
1041         }
1042         me.schedule = me._opts.transition((now - me._beginTime) / me._opts.duration);
1043 
1044         // render()接口，用来实现每个脉冲所要实现的效果
1045         // schedule 时间线的进度
1046         if (typeof me._opts.render == 'function') {
1047             me._opts.render(me.schedule);
1048         }
1049         // 执行下一个动作
1050         if (!me.terminative) {
1051             me._timer = setTimeout(function() {
1052                 me._launch()
1053             }, 1000 / me._opts.fps);
1054         }
1055     };

1057     /**
1058      * 变换效果函数库
1059      */
1060     var Transitions = {
1061         linear : function(t) {
1062             return t;
1063         },
1064         reverse : function(t) {
1065             return 1 - t;
1066         },
1067         easeInQuad : function(t) {
1068             return t * t;
1069         },
1070         easeInCubic : function(t) {
1071             return Math.pow(t, 3);
1072         },
1073         easeOutQuad : function(t) {
1074             return - (t * (t - 2));
1075         },
1076         easeOutCubic : function(t) {
1077             return Math.pow((t - 1), 3) + 1;
1078         },
1079         easeInOutQuad : function(t) {
1080             if (t < 0.5) {
1081                 return t * t * 2;
1082             } else {
1083                 return - 2 * (t - 2) * t - 1;
1084             }
1085             return;
1086         },
1087         easeInOutCubic : function(t) {
1088             if (t < 0.5) {
1089                 return Math.pow(t, 3) * 4;
1090             } else {
1091                 return Math.pow(t - 1, 3) * 4 + 1;
1092             }
1093         },
1094         easeInOutSine : function(t) {
1095             return (1 - Math.cos(Math.PI * t)) / 2;
1096         }
1097     };
